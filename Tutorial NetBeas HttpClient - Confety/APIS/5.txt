<?php
include'conexion.php';

$pdo=new Conexion();

if($_SERVER['REQUEST_METHOD']=='PUT'){
	$sql="UPDATE usuario SET nombre=:nombre WHERE id=:id";
	$stmt=$pdo->prepare($sql);
	$stmt->bindValue(':id',$_GET['id']);
	$stmt->bindValue(':nombre',$_GET['nombre']);
	$stmt->execute();
	header("HTTP/1.1 200 OK");
	exit;
	
}

?>
